#include <bits/stdc++.h>
using namespace std;

int lower_p(vector<pair<int, int> >& arr, pair<int, int>& p){
    auto low = lower_bound(arr.begin(), arr.end(), p);
    return low - arr.begin();
}

int upper_p(vector<pair<int, int> >& arr, pair<int, int>& p){
    auto up = upper_bound(arr.begin(), arr.end(), p);
    return up - arr.begin();
}

// second elemnt in vector
bool sort_by_second_element(const vector<int> &a,const vector<int> &b){
    return (a[1] < b[1]);
}

// second in pair
bool sortbysec(const pair<int,int> &a,
               const pair<int,int> &b){
    return (a.second < b.second);
}
