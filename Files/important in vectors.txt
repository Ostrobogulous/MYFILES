#include <bits/stdc++.h>
using namespace std;



bool sort_by_second_element(const vector<int> &a,const vector<int> &b){
    return (a[1] < b[1]);
}

vector <vector<int>> vect;
int main() {
    int x, y;
    cin >> x;
    for (int i = 0; i < x; i++){
        vector <int> o;
        for (int j = 0; j < x; j++){
            cin >> y;
            o.push_back(y);
        }
        vect.push_back(o);
    }
    vect[2][2] = 0;
    sort(vect.begin(), vect.end(), sort_by_second_element);
    for (int i = 0; i < x; i++){
        for (int j = 0; j < x; j++){
            cout << vect[i][j] << " ";
        }
        cout << endl;
    }
}
